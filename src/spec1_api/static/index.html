▷ Local Access
Terminal shortcuts for working with SPEC-1 directly
Read latest brief
cat ~/spec-1/briefs/spec1_brief_latest.md
List all briefs
ls ~/spec-1/briefs/spec1_brief_*.md
Start API server
cd ~/spec-1 && PYTHONPATH=src python -m spec1_api.main
Run intelligence cycle manually
cd ~/spec-1 && PYTHONPATH=src \
ANTHROPIC_API_KEY=$(grep ANTHROPIC_API_KEY .env | cut -d= -f2 | tr -d '\r') \
python -m spec1_engine.app.cycle
Run pytest
cd ~/spec-1 && PYTHONPATH=src pytest tests/ -v --tb=short
Build calibration proposal report
cd ~/spec-1 && PYTHONPATH=src python -m spec1_engine.tools.calibration_propose \
  --intel spec1_intelligence.jsonl \
  --verdicts verdicts.jsonl \
  --out-dir generated/
